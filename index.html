function renderAdzanMode(diff, waktu) { const adzan = document.getElementById('adzan-mode'), pText = document.getElementById('persiapan-text'), cText = document.getElementById('countdown-text'), pWrap = document.getElementById('progress-wrapper'), pBar = document.getElementById('progress-bar'), bgImg = document.getElementById('bg-adzan-img'); function validDurasi(nilai, fallback) { return Number.isFinite(nilai) && nilai > 0 ? nilai : fallback; } const durasiPersiapan = validDurasi(config.durasiPersiapan, 60); const durasiAdzan = validDurasi(config.durasiAdzan, 240); const durasiIqomah = validDurasi(config.durasiIqomah, waktu === 'subuh' ? 600 : 360); const durasiSholat = validDurasi(config.durasiSholat, 600); function tampilkanMode(mode, bgSrc, durasiZoom) { if (currentAdzanMode !== mode || adzan.style.display === 'none') { adzan.classList.remove('fade-in-anim', 'fade-out-anim'); bgImg.classList.remove('slow-zoom'); void adzan.offsetWidth; void bgImg.offsetWidth; bgImg.src = bgSrc || ""; bgImg.style.setProperty('--zoom-duration', durasiZoom + 's'); adzan.style.opacity = 0; adzan.style.display = 'flex'; adzan.classList.add('fade-in-anim'); bgImg.classList.add('slow-zoom'); currentAdzanMode = mode; fadeOutIqomahStarted = false; } } /* 1. MODE PERSIAPAN ADZAN Aktif sebelum adzan. Default 60 detik, atau mengikuti Durasi_Persiapan dari Sheet. */ if (diff > 0 && diff <= durasiPersiapan) { tampilkanMode('persiapan', config.bgPersiapan, durasiPersiapan); if (Math.floor(diff) >= 3 && Math.floor(diff) < 4 && !playedAdzan) { playAlert(); playedAdzan = true; } pText.innerText = "PERSIAPAN ADZAN"; pText.style.color = "#F5F1E8"; cText.innerText = Math.floor(diff); cText.style.color = "#E8D27A"; pWrap.style.display = 'none'; } /* 2. MODE SAAT ADZAN Aktif setelah waktu adzan masuk. Default durasi 240 detik, atau mengikuti Durasi_Adzan dari Sheet. */ else if (diff <= 0 && diff >= -durasiAdzan) { tampilkanMode('adzan', config.bgAdzan, durasiAdzan); playedAdzan = false; pText.innerText = "SAAT ADZAN HARAP TENANG"; pText.style.color = "#F5F1E8"; cText.innerText = ""; pWrap.style.display = 'flex'; const elapsed = Math.abs(diff); const persen = Math.min((elapsed / durasiAdzan) * 100, 100); pBar.style.width = persen + "%"; if (elapsed <= durasiAdzan / 3) { pBar.style.backgroundColor = '#FF0033'; } else if (elapsed <= (durasiAdzan / 3) * 2) { pBar.style.backgroundColor = '#FFFF00'; } else { pBar.style.backgroundColor = '#00C853'; } } /* 3. MODE IQOMAH Aktif setelah adzan selesai. Durasi mengikuti Durasi_Iqomah dari Sheet. Saat 3, 2, 1 detik terakhir: fade out. */ else if (diff < -durasiAdzan) { const sisaIqomah = durasiIqomah - (Math.abs(diff) - durasiAdzan); if (sisaIqomah > 0) { tampilkanMode('iqomah', config.bgIqomah, durasiIqomah); if (Math.floor(sisaIqomah) >= 3 && Math.floor(sisaIqomah) < 4 && !playedIqomah) { playAlert(); playedIqomah = true; } pText.innerHTML = '
IQOMAH
' + '
' + Math.floor(sisaIqomah / 60).toString().padStart(2, '0') + ':' + Math.floor(sisaIqomah % 60).toString().padStart(2, '0') + '
'; cText.innerText = ""; pWrap.style.display = 'none'; if (sisaIqomah <= 3 && !fadeOutIqomahStarted) { adzan.classList.remove('fade-in-anim'); void adzan.offsetWidth; adzan.classList.add('fade-out-anim'); fadeOutIqomahStarted = true; } } /* 4. MODE WAKTU SHOLAT Aktif setelah iqomah selesai. */ else if (sisaIqomah <= 0 && sisaIqomah > -durasiSholat) { tampilkanMode('sholat', config.bgSholat, durasiSholat); playedIqomah = false; pText.innerHTML = '
WAKTU SHOLAT
' + '
' + new Date().toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit', second: '2-digit', hour12: false }).replace(/\./g, ':') + '
'; cText.innerText = ""; pWrap.style.display = 'none'; } else { adzan.style.display = 'none'; adzan.style.opacity = 1; adzan.classList.remove('fade-in-anim', 'fade-out-anim'); bgImg.classList.remove('slow-zoom'); currentAdzanMode = ""; fadeOutIqomahStarted = false; } } }
